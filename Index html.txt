        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, collection, addDoc, onSnapshot, doc, setDoc, getDoc, setLogLevel, query } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        // --- GLOBAL APP STATE ---
        window.firebase = {};
        window.firebase.db = null;
        window.firebase.auth = null;
        window.firebase.userId = null;
        window.firebase.isAuthReady = false;
        
        // --- NEW SECURITY STATE FOR ADMIN ACCESS (MOCK) ---
        // CRITICAL: In a real app, this UID would be securely fetched after a successful login.
        // It is defined here only for demonstration purposes in this single file.
        const MOCK_ADMIN_UID = 'ADMIN-123-SECRET-UID'; 
        let isAdminAuthenticated = false; // Tracks successful mock login
        
        // 2. Navigation State

            heroImage2Url: 'https://placehold.co/640x480/D1FAE5/10B981?text=Mission+Image',
            heroImage3Url: 'https://placehold.co/640x480/FEE2E2/EF4444?text=Values+Image',
            registrationFee: 2800,

            lastUpdated: 'Loading...'

        // --- FIREBASE INITIALIZATION ---
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';
        const firebaseConfig = JSON.parse(typeof __firebase_config !== 'undefined' ? __firebase_config : '{}');
        const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;
        
        // setLogLevel('Debug'); // Uncomment for Firestore debugging

        if (Object.keys(firebaseConfig).length > 0) {
            const app = initializeApp(firebaseConfig);
            window.firebase.db = getFirestore(app);
            window.firebase.auth = getAuth(app);

            // Authentication setup
            onAuthStateChanged(window.firebase.auth, async (user) => {
                if (!user) {
                    try {
                        if (initialAuthToken) {
                            await signInWithCustomToken(window.firebase.auth, initialAuthToken);
                        } else {
                            await signInAnonymously(window.firebase.auth);
                        }
                    } catch (error) {
                        console.error("Firebase Sign-In Error:", error);
                    }
                }
                
                window.firebase.userId = window.firebase.auth.currentUser?.uid || crypto.randomUUID();
                window.firebase.isAuthReady = true;
                
                // Once authenticated, setup listeners and render
                setupSettingsListener(); // Load editable settings first
                if (contentArea) { // Check if DOMContentLoaded has run
                    renderPage(); 
                }
            });
            
            // Utility function to get the public collection reference (used for shared settings)
            window.getPublicCollectionRef = (collectionName) => {
                 if (!window.firebase.db) return null;
                 // Public path (for settings, counters, etc. - anything NOT sensitive)
                 return collection(window.firebase.db, `artifacts/${appId}/public/data/${collectionName}`);
            };

            // Utility function to get the PRIVATE collection reference (used for registration data)
            window.getPrivateCollectionRef = (collectionName) => {
                 if (!window.firebase.db || !window.firebase.userId) return null;
                 // PRIVATE path (for sensitive user data - only readable by that user)
                 return collection(window.firebase.db, `artifacts/${appId}/users/${window.firebase.userId}/${collectionName}`);
            };
            
            // Utility function to get the configuration document reference
            window.getConfigDocRef = () => {
                 if (!window.firebase.db) return null;
                 // Single document for app-wide settings
                 return doc(window.getPublicCollectionRef('settings'), 'config');
            };
            
        } else {
            console.warn("Firebase config is missing. Data persistence will be disabled.");
            window.firebase.isAuthReady = true; 
        }

        // --- ADMIN LOGIN MOCK FUNCTION ---
        window.attemptAdminLogin = function(event) {
            event.preventDefault();
            const inputElement = document.getElementById('admin-uid-input');
            const messageDiv = document.getElementById('admin-message-login');
            
            if (inputElement.value === MOCK_ADMIN_UID) {
                isAdminAuthenticated = true;

                messageDiv.textContent = 'Access Granted! (رسائی مل گئی!)';

                isAdminAuthenticated = false;
                messageDiv.className = 'p-3 rounded-md bg-red-100 text-red-700 font-semibold mt-4 shadow-md';
                messageDiv.textContent = 'Access Denied: Incorrect Admin UID. (رسائی سے انکار: غلط ایڈمن UID)';
            }

        // --- BACKEND READ OPERATION 1: APP SETTINGS ---
        function setupSettingsListener() {
            if (!window.firebase.db || !window.firebase.isAuthReady) return;

            const configRef = window.getConfigDocRef();
            
            // Live listener for app settings
            onSnapshot(configRef, (configDoc) => {
                if (configDoc.exists()) {
                    // Overwrite default settings with stored settings
                    const storedSettings = configDoc.data();
                    window.appSettings = { 
                        ...window.appSettings, 
                        ...storedSettings, 
                        lastUpdated: storedSettings.lastUpdated ? new Date(storedSettings.lastUpdated).toLocaleString() : 'Just Now'
                    };
                }
                // Re-render the current page to apply new settings
                renderPage();
            }, (error) => {
                console.error("Firestore Settings Listener Error:", error);
            });
        }
        
        // --- BACKEND WRITE OPERATION 1: SAVE APP SETTINGS (Admin Only) ---
        async function handleSettingsSave(event) {
            event.preventDefault();
            const messageDiv = document.getElementById('admin-message');
            messageDiv.className = 'p-3 rounded-md bg-yellow-100 text-yellow-700 font-semibold mt-4 shadow-md';
            messageDiv.textContent = 'Saving settings... (سیٹنگز محفوظ ہو رہی ہیں...)';
            
            if (!isAdminAuthenticated) {
                 messageDiv.textContent = 'Error: Authentication required to save settings. (ایرر: محفوظ کرنے کے لیے ایڈمن رسائی ضروری ہے)';
                 messageDiv.className = 'p-3 rounded-md bg-red-100 text-red-700 font-semibold mt-4 shadow-md';
                 return;
            }

            if (!window.firebase.db || !window.firebase.isAuthReady) {
                messageDiv.textContent = 'Error: Database not ready. Changes not saved. (ایرر: ڈیٹا بیس تیار نہیں)';
                messageDiv.className = 'p-3 rounded-md bg-red-100 text-red-700 font-semibold mt-4 shadow-md';
                return;
            }

            try {
                const newSettings = {
                    heroImage1Url: document.getElementById('setting-img1').value,
                    heroImage2Url: document.getElementById('setting-img2').value,
                    heroImage3Url: document.getElementById('setting-img3').value,
                    registrationFee: parseFloat(document.getElementById('setting-fee').value),
                    serviceHours: document.getElementById('setting-hours').value,
                    lastUpdated: new Date().toISOString()
                };

                const configRef = window.getConfigDocRef();
                // *** SET DOCUMENT OPERATION (Overwrite existing config) ***
                await setDoc(configRef, newSettings);

                messageDiv.textContent = 'Settings successfully updated! (سیٹنگز کامیابی سے اپ ڈیٹ ہو گئیں!)';
                messageDiv.className = 'p-3 rounded-md bg-secondary text-white font-bold mt-4 shadow-lg';
                
            } catch (error) {
                console.error("Error saving settings:", error);
                messageDiv.textContent = `Error: Failed to save settings. (ایرر: سیٹنگز محفوظ نہیں ہو سکیں) ${error.message}`;
                messageDiv.className = 'p-3 rounded-md bg-red-100 text-red-700 font-semibold mt-4 shadow-md';
            }
        }

        async function handleRegistrySubmit(event) {
            event.preventDefault();

            const country = document.getElementById('country').value.trim();

            // 2. Fetch all file inputs (Files are not truly uploaded, just names are tracked)
            const feeSlipFile = document.getElementById('fee-slip').files[0]; 
            const cnicFrontFile = document.getElementById('cnic-front').files[0]; 
            const cnicBackFile = document.getElementById('cnic-back').files[0];   
            const passportPhotoFile = document.getElementById('passport-photo').files[0]; 
            
            // 3. Validation
            if (!fullName || !email || !cnicNumber || !phoneNumber || !homeAddress || !city || !country) {
                messageDiv.textContent = 'Please fill in all required text fields. (براہ کرم تمام مطلوبہ فیلڈز مکمل کریں!)';
                messageDiv.className = 'p-3 rounded-md bg-red-100 text-red-700 font-semibold mt-4 shadow-md';
                return;
            }
            if (!feeSlipFile || !cnicFrontFile || !cnicBackFile || !passportPhotoFile) {
                messageDiv.textContent = 'Please upload all required documents. (براہ کرم تمام مطلوبہ دستاویزات اپ لوڈ کریں!)';
                messageDiv.className = 'p-3 rounded-md bg-red-100 text-red-700 font-semibold mt-4 shadow-md';
                return;
            }

            if (window.firebase.db && window.firebase.isAuthReady && window.firebase.userId) {
                try {
                    const registrationData = {
                        // Public/Searchable Data (Can be read by app settings if needed)
                        fullName: fullName,
                        city: city,
                        country: country,
                        registrationFee: window.appSettings.registrationFee, 
                        registeredAt: new Date().toISOString(),
                        userId: window.firebase.userId,
                        
                        // SENSITIVE/PRIVATE DATA (Saved in User's Private Collection)
                        email: email,
                        cnicNumber: cnicNumber, 
                        phoneNumber: phoneNumber, 
                        homeAddress: homeAddress, 
                        
                        // Document File Names (Proof of upload)
                        feeSlipFileName: feeSlipFile.name,
                        cnicFrontFileName: cnicFrontFile.name, 
                        cnicBackFileName: cnicBackFile.name,   
                        passportPhotoFileName: passportPhotoFile.name, 
                    };

                    // *** CRITICAL SECURITY CHANGE: Saving to PRIVATE collection ***
                    // The path is: /artifacts/{appId}/users/{userId}/registrations/{documentId}
                    const registrationsRef = window.getPrivateCollectionRef('registrations');
                    await addDoc(registrationsRef, registrationData);

                    messageDiv.textContent = 'Registration Successful! Your sensitive data is saved securely under your User ID. (رجسٹریشن کامیاب! آپ کا حساس ڈیٹا محفوظ طریقے سے محفوظ ہے)';
                    messageDiv.className = 'p-3 rounded-md bg-secondary text-white font-bold mt-4 shadow-lg';
                    form.reset();
                    
                } catch (error) {
                    console.error("Error submitting registration to Firestore (Backend Write Error):", error);
                    messageDiv.textContent = `Error: Registration failed. (رجسٹریشن ناکام) ${error.message}`;
                    messageDiv.className = 'p-3 rounded-md bg-red-100 text-red-700 font-semibold mt-4 shadow-md';
                }
            } else {
                messageDiv.textContent = 'Registration Successful (Data Not Saved: Firebase Not Ready or User ID Missing)!';
                messageDiv.className = 'p-3 rounded-md bg-yellow-100 text-yellow-700 font-semibold mt-4 shadow-md';
                form.reset();
            }
        }

        let unsubscribeRegistrationListener = null;
        const BASE_REGISTRATIONS = 50000;

        function setupRegistrationListener() {
            if (unsubscribeRegistrationListener) {
                unsubscribeRegistrationListener();
                unsubscribeRegistrationListener = null;

            const remainingElement = document.getElementById('remaining-slots');

                if (counterElement) counterElement.textContent = '...';
                if (remainingElement) remainingElement.textContent = '...';
                return; 
            }

            try {
                // We use the 'settings' collection count as a mock for public activity
                const publicRef = window.getPublicCollectionRef('public_activity_mock');
                
                // *** ON SNAPSHOT OPERATION (Live Backend Read on a non-sensitive public collection) ***
                // Note: We use a simple query on a public collection to safely demonstrate the counter feature.
                unsubscribeRegistrationListener = onSnapshot(publicRef, (snapshot) => {
                    const actualFirestoreCount = snapshot.size;
                    
                    const displayedTotal = BASE_REGISTRATIONS + actualFirestoreCount;
                    let remaining = MAX_REGISTRATIONS - displayedTotal;
                    if (remaining < 0) remaining = 0; 

                    counterElement.textContent = displayedTotal.toLocaleString('en-US');
                    remainingElement.textContent = remaining.toLocaleString('en-US');

                }, (error) => {
                    console.error("Firestore Listener Error (Backend Read Error):", error);
                    counterElement.textContent = 'Error!';
                    remainingElement.textContent = 'Error!';
                });
            } catch (e) {
                console.error("Failed to setup listener:", e);
                counterElement.textContent = 'Error!';
                remainingElement.textContent = 'Error!';
            }
        }
